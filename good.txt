✅ STEP 11: Add to Pipeline (Final Integration)

Update your Jenkinsfile:

pipeline {
    agent any

    tools {
        maven 'Maven3'
    }

    stages {

        stage('CHECKOUT') {
            steps {
                git branch: 'main', url: 'https://github.com/Shreya-Rajashekara12/demo12.git'
            }
        }

        stage('BUILD') {
            steps {
                dir('demo12') {
                    bat 'mvn clean install'
                }
            }
        }

        stage('TEST') {
            steps {
                dir('demo12') {
                    bat 'mvn test'
                }
            }
        }
    }

    post {
        success {
            slackSend channel: '#jenkins', message: "✅ SUCCESS: ${env.JOB_NAME} #${env.BUILD_NUMBER}"
        }
        failure {
            slackSend channel: '#jenkins', message: "❌ FAILED: ${env.JOB_NAME} #${env.BUILD_NUMBER}"
        }
    }
}

o Set:

o Debugger for Java
